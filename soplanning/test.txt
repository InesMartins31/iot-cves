sudo apt-get update

sudo systemctl start apache2
sudo systemctl enable apache2
sudo systemctl start mariadb
sudo systemctl enable mariadb

sudo mysql -u root <<MYSQL_SCRIPT
USE mysql;
DROP USER root;
FLUSH PRIVILEGES;
CREATE USER 'root'@'localhost' IDENTIFIED BY '';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'localhost';
UPDATE user SET plugin='auth_socket' WHERE User='root';
FLUSH PRIVILEGES;
exit;
MYSQL_SCRIPT

sudo systemctl restart mariadb

mysql -u root <<MYSQL_SCRIPT
CREATE DATABASE soplanning;
CREATE USER 'root'@'localhost' IDENTIFIED BY 'root';
GRANT ALL PRIVILEGES ON soplanning.* TO 'soplanning'@'localhost';
FLUSH PRIVILEGES;
MYSQL_SCRIPT
sudo mkdir -p /var/www/html/soplanning
cd /tmp
wget https://netcologne.dl.sourceforge.net/project/soplanning/soplanning/v1.45/soplanning-1-45.zip
unzip soplanning-1-45.zip
sudo cp -r soplanning /var/www/html/soplanning
sudo chown -R www-data.www-data /var/www/html/soplanning
sudo chmod -R 775 /var/www/html/soplanning
